CREATE TABLE Departments (
    Dept_ID INT PRIMARY KEY,
    Dept_Name VARCHAR(50)
);

(4,'Sales');
CREATE TABLE Employee1 (
    Emp_ID INT PRIMARY KEY,
    Emp_Name VARCHAR(50),
    Dept_ID INT,
    Salary DECIMAL(10,2),
    FOREIGN KEY (Dept_ID) REFERENCES Departments(Dept_ID)
);

SELECT
    Employee1.Emp_ID,
    Employee1.Emp_Name,
    Departments.Dept_Name,
    Employee1.Salary
FROM Employee1
INNER JOIN Departments
ON Employee1.Dept_ID = Departments.Dept_ID;
SELECT
    Employee1.Emp_Name,
    Departments.Dept_Name
FROM Employee1
LEFT JOIN Departments
ON Employee1.Dept_ID = Departments.Dept_ID;
SELECT
    Employee1.Emp_Name,
    Departments.Dept_Name
FROM Employee1
RIGHT JOIN Departments
ON Employee1.Dept_ID = Departments.Dept_ID;
SELECT
    Employee1.Emp_Name,
    Departments.Dept_Name
FROM Employee1
CROSS JOIN Departments;
SELECT
    Emp_Name,
    Salary
FROM Employee1
WHERE Salary > 50000;
SELECT
    Employee1.Emp_Name,
    Departments.Dept_Name
FROM Employee1
INNER JOIN Departments
ON Employee1.Dept_ID = Departments.Dept_ID
WHERE Departments.Dept_Name='IT';

SELECT
Departments.Dept_Name,
COUNT(Employee1.Emp_ID) AS TotalEmployee1
FROM Employee1
INNER JOIN Departments
ON Employee1.Dept_ID=Departments.Dept_ID
GROUP BY Departments.Dept_Name;
SELECT
Departments.Dept_Name,
MAX(Employee1.Salary) AS HighestSalary
FROM Employee1
INNER JOIN Departments
ON Employee1.Dept_ID=Departments.Dept_ID
GROUP BY Departments.Dept_Name;
